Задание 1.

SELECT "Couriers"."login",
	COUNT("Orders"."track")
FROM "Orders"
JOIN "Couriers" ON "Couriers"."id" = "Orders"."courierId"
GROUP BY "Couriers"."login",
	"Orders"."inDelivery"
HAVING "Orders"."inDelivery" = true;

Задание 2.

SELECT "track",
	CASE
		WHEN "finished" = true THEN '2'
		WHEN "cancelled" = true THEN '-1'
		WHEN "inDelivery" = true THEN '1'
		ELSE '0'
	END AS "status"
FROM "Orders";
